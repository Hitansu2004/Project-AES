-- ============================================================
-- USERS & AUTH
-- ============================================================

CREATE TABLE users (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    phone_number        VARCHAR(15) NOT NULL UNIQUE,          -- +91XXXXXXXXXX format
    name                VARCHAR(100),
    email               VARCHAR(150),
    role                VARCHAR(20) NOT NULL DEFAULT 'CUSTOMER',
                        -- CUSTOMER | CRM_AGENT | SERVICE_MANAGER | ADMIN
    is_active           BOOLEAN NOT NULL DEFAULT TRUE,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE otp_tokens (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    phone_number        VARCHAR(15) NOT NULL,
    otp_code            VARCHAR(6) NOT NULL,
    expires_at          TIMESTAMPTZ NOT NULL,
    is_used             BOOLEAN NOT NULL DEFAULT FALSE,
    attempt_count       INT NOT NULL DEFAULT 0,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE refresh_tokens (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id             UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    token               VARCHAR(512) NOT NULL UNIQUE,
    expires_at          TIMESTAMPTZ NOT NULL,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ============================================================
-- PROPERTIES & AC ASSETS
-- ============================================================

CREATE TABLE properties (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    customer_id         UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    label               VARCHAR(100) NOT NULL,               -- "Villa #42", "Office - Floor 3"
    address_line1       VARCHAR(200) NOT NULL,
    address_line2       VARCHAR(200),
    city                VARCHAR(100) NOT NULL DEFAULT 'Hyderabad',
    pincode             VARCHAR(10),
    property_type       VARCHAR(20) NOT NULL DEFAULT 'RESIDENTIAL',
                        -- RESIDENTIAL | COMMERCIAL | HOSPITAL | HOTEL
    is_primary          BOOLEAN NOT NULL DEFAULT FALSE,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE ac_units (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    property_id         UUID NOT NULL REFERENCES properties(id) ON DELETE CASCADE,
    customer_id         UUID NOT NULL REFERENCES users(id),
    room_label          VARCHAR(100) NOT NULL,               -- "Master Bedroom"
    ac_type             VARCHAR(20) NOT NULL,
                        -- SPLIT | CASSETTE | CENTRAL | VRF_VRV | WINDOW | PORTABLE
    brand               VARCHAR(50) NOT NULL,               -- "Daikin", "Voltas", etc.
    model_number        VARCHAR(100),                       -- "FTKF35TV"
    tonnage             DECIMAL(3,1) NOT NULL,              -- 1.5, 2.0, etc.
    energy_star_rating  INT,                                -- 3, 4, 5
    installation_date   DATE,
    warranty_expiry     DATE,
    amc_contract_id     UUID,                               -- FK added after amc_contracts table
    warranty_status     VARCHAR(20) NOT NULL DEFAULT 'UNKNOWN',
                        -- IN_WARRANTY | EXPIRED | UNKNOWN
    service_status      VARCHAR(20) NOT NULL DEFAULT 'P3_PAID',
                        -- P1_AMC | P2_WARRANTY | P3_PAID
    is_active           BOOLEAN NOT NULL DEFAULT TRUE,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ============================================================
-- AMC CONTRACTS
-- ============================================================

CREATE TABLE amc_contracts (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    customer_id         UUID NOT NULL REFERENCES users(id),
    property_id         UUID NOT NULL REFERENCES properties(id),
    contract_number     VARCHAR(50) NOT NULL UNIQUE,        -- "AMC-2024-0042"
    start_date          DATE NOT NULL,

    visits_completed    INT NOT NULL DEFAULT 0,

    contract_value      DECIMAL(10,2),
    notes               TEXT,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Add FK from ac_units to amc_contracts
ALTER TABLE ac_units ADD CONSTRAINT fk_ac_units_amc
    FOREIGN KEY (amc_contract_id) REFERENCES amc_contracts(id);

CREATE TABLE amc_visits (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    contract_id         UUID NOT NULL REFERENCES amc_contracts(id),
    visit_number        INT NOT NULL,                       -- 1, 2, 3, 4
    scheduled_date      DATE,
    scheduled_time_slot VARCHAR(20),                        -- "MORNING" | "AFTERNOON" | "EVENING"
    actual_visit_date   TIMESTAMPTZ,
    engineer_id         UUID REFERENCES users(id),
    status              VARCHAR(20) NOT NULL DEFAULT 'SCHEDULED',
                        -- SCHEDULED | COMPLETED | MISSED | RESCHEDULED
    notes               TEXT,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ============================================================
-- INSTALLATION REQUESTS
-- ============================================================

CREATE TABLE installation_requests (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    request_number      VARCHAR(20) NOT NULL UNIQUE,        -- "REQ-2024-1089"
    customer_id         UUID NOT NULL REFERENCES users(id),
    property_id         UUID REFERENCES properties(id),
    property_address    TEXT,                               -- free text if new address
    ac_type             VARCHAR(20) NOT NULL,
    brand               VARCHAR(50),
    model_number        VARCHAR(100),
    tonnage             DECIMAL(3,1),
    energy_rating       INT,
    rooms_json          JSONB,
    -- [{room_type: "Master Bedroom", size_sqft: 180, ac_type: "SPLIT"}]
    scheduled_date      DATE,
    scheduled_slot      VARCHAR(20),                        -- "MORNING" | "AFTERNOON" | "EVENING"
    status              VARCHAR(30) NOT NULL DEFAULT 'PENDING',
                        -- PENDING | CONFIRMED | SITE_VISIT_DONE | QUOTE_SENT
                        -- | QUOTE_ACCEPTED | INSTALLATION_SCHEDULED | COMPLETED | CANCELLED
    assigned_engineer_id UUID REFERENCES users(id),
    estimated_cost      DECIMAL(10,2),
    notes               TEXT,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ============================================================
-- SERVICE TICKETS — CORE TABLE
-- ============================================================

CREATE TABLE service_tickets (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    ticket_number       VARCHAR(20) NOT NULL UNIQUE,        -- "TKT-2024-0089"
    customer_id         UUID NOT NULL REFERENCES users(id),
    property_id         UUID NOT NULL REFERENCES properties(id),
    ac_unit_id          UUID NOT NULL REFERENCES ac_units(id),
    priority            VARCHAR(5) NOT NULL,                -- "P1" | "P2" | "P3"
    service_type        VARCHAR(20) NOT NULL,               -- "AMC" | "WARRANTY" | "PAID"
    problem_category    VARCHAR(30) NOT NULL,
                        -- NOT_COOLING | NOISE | LEAKING | NOT_TURNING_ON | NO_AIRFLOW | REMOTE_WIFI | OTHER
    error_code          VARCHAR(10),                        -- "E1", "H6", etc.
    problem_description TEXT,
    photos_json         JSONB,                              -- array of S3/file URLs
    scheduled_date      DATE,
    scheduled_slot      VARCHAR(20),

    -- Current assignment
    current_level       INT NOT NULL DEFAULT 1,             -- 1 | 2 | 3
    current_assignee_id UUID REFERENCES users(id),          -- CRM agent / service manager / admin
    assigned_at         TIMESTAMPTZ,

    -- Status
    status              VARCHAR(20) NOT NULL DEFAULT 'OPEN',
                        -- OPEN | ACKNOWLEDGED | ASSIGNED | IN_PROGRESS | RESOLVED | CLOSED | CANCELLED

    -- SLA tracking
    sla_deadline_l1     TIMESTAMPTZ,   -- created_at + 30 min
    sla_deadline_l2     TIMESTAMPTZ,   -- escalated_to_l2_at + 60 min
    sla_deadline_final  TIMESTAMPTZ,   -- based on priority: P1=4hr, P2=8hr, P3=24hr from created_at
    acknowledged_at     TIMESTAMPTZ,
    resolved_at         TIMESTAMPTZ,
    closed_at           TIMESTAMPTZ,

    -- Charges (P3 only)
    estimated_charge    DECIMAL(10,2),
    final_charge        DECIMAL(10,2),
    charge_accepted     BOOLEAN,

    -- Rating
    customer_rating     INT,           -- 1-5 stars
    customer_feedback   TEXT,

    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ============================================================
-- TICKET ESCALATION HISTORY
-- ============================================================

CREATE TABLE ticket_escalation_log (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    ticket_id           UUID NOT NULL REFERENCES service_tickets(id),
    from_level          INT NOT NULL,                       -- 1 | 2 | 3
    to_level            INT NOT NULL,
    from_user_id        UUID REFERENCES users(id),
    reason              VARCHAR(200) NOT NULL,
                        -- "Auto: 30min timeout" | "Manual escalation by agent"
    escalation_type     VARCHAR(10) NOT NULL,               -- "AUTO" | "MANUAL"
    escalated_at        TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ============================================================
-- TICKET ACTIVITY TIMELINE
-- ============================================================

CREATE TABLE ticket_activities (
    id                  UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

    user_id             UUID REFERENCES users(id),          -- null = system

                        -- | NOTE_ADDED | PHOTO_ADDED | STATUS_CHANGED | RESOLVED | RATED
    description         TEXT NOT NULL,
    metadata_json       JSONB,                              -- extra data if needed
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

─────────────────────────────────────────────────────────────────────────────
4.1 AUTH APIs
─────────────────────────────────────────────────────────────────────────────

POST /api/v1/auth/send-otp
  Purpose:  Send OTP to customer phone number
  Public:   Yes
  Body:     { "phoneNumber": "+919876543210" }
  Logic:
    1. Validate phone format (+91 followed by 10 digits)
    2. Check rate limit: max 3 OTPs per phone per 10 minutes
    3. Generate 6-digit OTP
    4. Save to otp_tokens with expires_at = NOW() + 10 minutes
    5. In demo mode: log OTP to console + return it in response (field "otpForDemo")
    6. In production: call Twilio SMS API
  Response: { "success": true, "data": { "expiresInSeconds": 600, "otpForDemo": "123456" } }

POST /api/v1/auth/verify-otp
  Purpose:  Verify OTP and return JWT tokens
  Public:   Yes
  Body:     { "phoneNumber": "+919876543210", "otp": "123456" }
  Logic:
    1. Find latest unused OTP for phone where is_used=false and expires_at > NOW()
    2. Increment attempt_count; if > 5, invalidate OTP and return error
    3. Compare OTP code (case-insensitive)
    4. Mark OTP as used
    5. Find or create user record with role=CUSTOMER
    6. Generate JWT access token (15 min expiry) + refresh token (7 days)
    7. Save refresh token to refresh_tokens table
  Response: { "success": true, "data": { "accessToken": "...", "refreshToken": "...", "user": { "id": "...", "name": "...", "role": "CUSTOMER" } } }

POST /api/v1/auth/staff-login
  Purpose:  Login for CRM agents, managers, admins (password-based)
  Public:   Yes
  Body:     { "phoneNumber": "+919800000001", "password": "password123" }
  Note:     Staff passwords stored hashed (BCrypt). Add password field to users table in V3 migration.
  Response: Same as verify-otp response

POST /api/v1/auth/refresh
  Purpose:  Get new access token using refresh token
  Public:   Yes
  Body:     { "refreshToken": "..." }
  Response: { "success": true, "data": { "accessToken": "..." } }

POST /api/v1/auth/logout
  Purpose:  Invalidate refresh token
  Auth:     Required
  Body:     { "refreshToken": "..." }
  Logic:    Delete refresh token record

4.8 ESCALATION ENGINE — SCHEDULED JOB
─────────────────────────────────────────────────────────────────────────────

This is the MOST CRITICAL backend component. Implement as a Spring @Scheduled job.

CLASS: EscalationEngineService.java

@Scheduled(fixedDelay = 30000)  // Run every 30 seconds
public void checkAndEscalate() {
    // Find tickets where L1 SLA is breached and still at level 1
    List<ServiceTicket> l1Overdue = ticketRepository.findL1OverdueTickets(LocalDateTime.now());
    for (ServiceTicket ticket : l1Overdue) {
        escalateToL2(ticket, "Auto: 30min L1 timeout");
    }

    // Find tickets where L2 SLA is breached and still at level 2
    List<ServiceTicket> l2Overdue = ticketRepository.findL2OverdueTickets(LocalDateTime.now());
    for (ServiceTicket ticket : l2Overdue) {
        escalateToL3(ticket, "Auto: 60min L2 timeout");
    }
}

private void escalateToL2(ServiceTicket ticket, String reason) {
    int previousLevel = ticket.getCurrentLevel();

    // Log escalation
    TicketEscalationLog log = new TicketEscalationLog();
    log.setTicket(ticket);
    log.setFromLevel(1);
    log.setToLevel(2);
    log.setFromUserId(ticket.getCurrentAssigneeId());
    log.setReason(reason);
    log.setEscalationType("AUTO");
    log.setEscalatedAt(LocalDateTime.now());
    escalationLogRepository.save(log);

    // Update ticket
    ticket.setCurrentLevel(2);
    ticket.setSlaDeadlineL2(LocalDateTime.now().plusMinutes(60));

    // Assign to available SERVICE_MANAGER (round-robin)
    User manager = assignmentService.getNextAvailableManager();
    ticket.setCurrentAssigneeId(manager.getId());
    ticket.setAssignedAt(LocalDateTime.now());
    ticket.setUpdatedAt(LocalDateTime.now());
    ticketRepository.save(ticket);

    // Create activity
    activityService.addActivity(ticket.getId(), null,
        "ESCALATED", "Auto-escalated to Level 2 (Service Managers). L1 did not respond within 30 minutes.");

    // Send notifications
    notificationService.notifyCustomer(ticket.getCustomerId(),
        "Update on ticket " + ticket.getTicketNumber(),
        "Your request has been escalated to our Service Manager team for faster resolution.");
    notificationService.notifyUser(manager.getId(),
        "New Escalated Ticket",
        "Ticket " + ticket.getTicketNumber() + " has been escalated to you.");

    // WebSocket broadcast
    webSocketService.broadcastTicketUpdate(ticket.getTicketNumber(),
        Map.of("level", 2, "status", "ESCALATED_TO_L2"));
}

}

================================================================================
SECTION 6 — BUSINESS RULES & SLA DEFINITIONS
================================================================================

SLA RESPONSE TIMES (from ticket creation):
  P1 (AMC):          4 hours total resolution SLA
  P2 (Warranty):     8 hours total resolution SLA
  P3 (Paid):         24 hours total resolution SLA

L1 AUTO-ESCALATION TRIGGER:
  If ticket at Level 1 AND acknowledged_at IS NULL AND NOW() > sla_deadline_l1
  (sla_deadline_l1 = created_at + 30 minutes)
  → Auto-escalate to Level 2 with reason "Auto: 30min L1 timeout"

L2 AUTO-ESCALATION TRIGGER:
  If ticket at Level 2 AND NOW() > sla_deadline_l2
  (sla_deadline_l2 = escalated_to_l2_at + 60 minutes)
  → Auto-escalate to Level 3 with reason "Auto: 60min L2 timeout"

MANUAL ESCALATION:
  L1 agent CAN escalate to L2 manually at any time
  L2 manager CAN escalate to L3 manually
  Customer CANNOT escalate manually (only AES staff can)

ACKNOWLEDGMENT:
  L1 agent clicking "Mark Acknowledged" stops the L1 SLA breach countdown
  It does NOT resolve the ticket — ticket remains ACKNOWLEDGED until
  either "Assign Engineer" or "Resolve" is clicked

PRIORITY DETERMINATION (automatic, based on ac_unit.service_status):
  P1_AMC        → service_type = "AMC",       priority = "P1"
  P2_WARRANTY   → service_type = "WARRANTY",  priority = "P2"
  P3_PAID       → service_type = "PAID",       priority = "P3"

TICKET NUMBER FORMAT:
  TKT-{YYYY}-{4-digit-zero-padded-sequence}
  Sequence is per-year, starts at 0001 each calendar year
  Use a DB sequence: CREATE SEQUENCE ticket_seq_2024 START 1;

INSTALLATION REQUEST NUMBER FORMAT:
  REQ-{YYYY}-{4-digit-sequence}

================================================================================
SECTION 7 — ERROR HANDLING & VALIDATION
================================================================================

PHONE NUMBER VALIDATION:
  Must match regex: ^\+91[6-9]\d{9}$
  (Indian mobile numbers: +91 followed by 6/7/8/9 and 9 more digits)

OTP RATE LIMITING:
  Max 3 OTP requests per phone per 10 minutes
  Max 5 verification attempts per OTP
  OTP validity: 10 minutes

JWT TOKEN:
  Access token: 15 minutes
  Refresh token: 7 days
  Algorithm: HS256
  Claims: sub (userId), role, iat, exp

TICKET CREATION VALIDATION:
  - ac_unit_id must belong to authenticated customer
  - scheduled_date must be tomorrow or later
  - scheduled_slot must be MORNING | AFTERNOON | EVENING
  - photos_json array max 4 items

HTTP ERROR CODES:
  400 BAD_REQUEST      → Validation failures
  401 UNAUTHORIZED     → Missing/invalid JWT
  403 FORBIDDEN        → Correct JWT but wrong role
  404 NOT_FOUND        → Resource not found
  409 CONFLICT         → e.g., duplicate request
  422 UNPROCESSABLE    → Business rule violation
  429 TOO_MANY_REQUESTS → OTP rate limit hit
  500 INTERNAL_ERROR   → Unexpected server error

ALL 500 errors: log stack trace, return generic message to client (never expose internals)

SECTION 9 — KEY COMPONENT IMPLEMENTATIONS
================================================================================

─── SlaCountdown Component ───
// components/ui/SlaCountdown.tsx
// Props: deadlineISO: string, onBreach?: () => void
// Calculates remaining seconds from deadline
// Updates every second via setInterval
// Color: green > 20min, orange 10-20min, red < 10min
// Format: "24 min" | "9 min 45 sec" | "BREACHED"
// On component unmount: clear interval

─── EscalationLadder Component ───
// components/ticket/EscalationLadder.tsx
// Props: currentLevel: 1|2|3, status: string, slaRemainingSeconds: number
// Renders 3-step vertical stepper
// Level states: ACTIVE (pulsing teal border), STANDBY (gray), ESCALATED_PAST (gray + X icon)
// WebSocket subscription via useTicketUpdates hook
// Re-renders instantly on escalation event

─── useSlaCountdown Hook ───
// hooks/useSlaCountdown.ts
// Input: deadlineISO string
// Output: { remainingSeconds: number, isBreached: boolean, displayText: string }
// Uses setInterval(1000) internally
// Cleans up on unmount

─── STOMP Client ───
// lib/websocket/stompClient.ts
// Singleton STOMP client over SockJS at /ws
// Auto-reconnect on disconnect
// subscribe(topic, callback) method
// unsubscribe(subscriptionId) method
// Export: stompClient instance

─── Axios Interceptors ───
// lib/api/axios.ts
// Request interceptor: add Authorization: Bearer {accessToken}
// Response interceptor (401): attempt token refresh via /auth/refresh
// If refresh succeeds: retry original request with new token
// If refresh fails: clear auth store → redirect to /login

WebSocket Config:
  @Configuration
  @EnableWebSocketMessageBroker
  public class WebSocketConfig implements WebSocketMessageBrokerConfigurer {
      @Override
      public void registerStompEndpoints(StompEndpointRegistry registry) {
          registry.addEndpoint("/ws").setAllowedOriginPatterns("*").withSockJS();
      }
      @Override
      public void configureMessageBroker(MessageBrokerRegistry config) {
          config.enableSimpleBroker("/topic");
          config.setApplicationDestinationPrefixes("/app");
      }
  }
